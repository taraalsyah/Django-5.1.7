>install Python 3.11 , create Virtual env , install depedencies=

# Download Python 3.11
cd /tmp
wget https://www.python.org/ftp/python/3.11.9/Python-3.11.9.tgz

# Extract
tar xzf Python-3.11.9.tgz
cd Python-3.11.9

# Compile dan install
./configure --enable-optimizations
make altinstall

# Cek versi
python3.11 --version

# Download OpenSSL 1.1.1
cd /tmp
wget https://www.openssl.org/source/openssl-1.1.1w.tar.gz

# Extract
tar xzf openssl-1.1.1w.tar.gz
cd openssl-1.1.1w

# Compile dan install
./config --prefix=/usr/local/openssl --openssldir=/usr/local/openssl shared zlib
make
sudo make install

#Setelah OpenSSL terinstall, kompile ulang Python 3.11:
cd /tmp/Python-3.11.9

# Clean hasil compile sebelumnya
make clean

# Configure ulang dengan OpenSSL baru
./configure --enable-optimizations --with-openssl=/usr/local/openssl

# Compile dan install
make altinstall

#Coba cek dulu OpenSSL-nya terinstall dengan benar:
/usr/local/openssl/bin/openssl version

#Lalu cek library-nya ada:
ls /usr/local/openssl/lib/

#Kalau ada, tambahkan ke library path lalu compile ulang:
# Tambah library path
export LD_LIBRARY_PATH=/usr/local/openssl/lib:$LD_LIBRARY_PATH
export PKG_CONFIG_PATH=/usr/local/openssl/lib/pkgconfig

# Clean dan compile ulang
cd /tmp/Python-3.11.9
make clean

./configure --enable-optimizations --with-openssl=/usr/local/openssl --with-openssl-rpath=/usr/local/openssl/lib

make altinstall

# Tambahkan library OpenSSL ke sistem
echo "/usr/local/openssl/lib" | sudo tee /etc/ld.so.conf.d/openssl.conf

# Refresh library cache
sudo ldconfig

# Cek lagi
/usr/local/openssl/bin/openssl version

#Kalau berhasil, lanjut compile ulang Python:
export LD_LIBRARY_PATH=/usr/local/openssl/lib:$LD_LIBRARY_PATH

cd /tmp/Python-3.11.9
make clean

./configure --enable-optimizations --with-openssl=/usr/local/openssl --with-openssl-rpath=/usr/local/openssl/lib

make altinstall

# Cek SSL
python3.11 -c "import ssl; print(ssl.OPENSSL_VERSION)"

#compile ulang=
cd /tmp/Python-3.11.9
make clean

./configure --with-openssl=/usr/local/openssl --with-openssl-rpath=/usr/local/openssl/lib 2>&1 | grep -i ssl

# Cek library openssl ada
ls /usr/local/openssl/lib/

# Cek header openssl ada
ls /usr/local/openssl/include/openssl/

#Sekarang compile ulang:
sudo make altinstall

#Sekarang cek SSL dulu:
python3.11 -c "import ssl; print(ssl.OPENSSL_VERSION)"
